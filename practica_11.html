Sistema de Gestión de Notas — Curso ESPAM

arreglos unidimensionales · funciones con const double[] · estadísticas · burbuja · búsqueda lineal

Duración180 min (3h)
ModalidadIndividual → Parejas
Entregable: .cpp + trazas + bitácora
F1 · Diseño F2 · Pseudocódigo F3 · C++ F4 · Extensión F5 · Análisis IA F6 · Bitácora
Objetivo de la Sesión
El estudiante construye un sistema modular de gestión de notas que almacena las calificaciones de 10 estudiantes en un arreglo, calcula estadísticas (mín, máx, promedio, aprobados), ordena con burbuja, y permite búsqueda por nota. Cada operación es una función independiente con la firma correcta (const double[] para lectura, double[] para modificación).
🎓 Contexto del Problema
La cátedra de Introducción a la Programación de ESPAM MFL necesita un sistema de consola para procesar las notas del primer parcial de 10 estudiantes. El sistema debe: 1) ingresar notas con validación [0–10] 2) calcular estadísticas completas 3) listar aprobados y reprobados 4) mostrar el ranking ordenado de mayor a menor 5) buscar si una nota específica existe en la lista.

Implementación Completa en C++ — Individual

C++ · Sistema de notas ESPAM — implementación completa
sistema_notas.cpp 
#include <iostream>
#include <iomanip>
using namespace std;

const int N = 10;

// ─── PROTOTIPOS ─────────────────────────────────────────────────
void   ingresar(double notas[], int n);
double calcularProm(const double notas[], int n);
double calcularMin(const double notas[], int n);
double calcularMax(const double notas[], int n);
int    contarAprobados(const double notas[], int n);
void   mostrarLista(const double notas[], int n);
void   ordenarDesc(double notas[], int n);
int    buscarNota(const double notas[], int n, double obj);

int main() {
    double notas[N];
    ingresar(notas, N);

    cout << "
══════ REPORTE DE NOTAS ══════
";
    mostrarLista(notas, N);
    cout << fixed << setprecision(2);
    cout << "Promedio : " << calcularProm(notas,N) << endl;
    cout << "Máxima   : " << calcularMax(notas,N)  << endl;
    cout << "Mínima   : " << calcularMin(notas,N)  << endl;
    cout << "Aprobados: " << contarAprobados(notas,N) << "/" << N << endl;

    cout << "
── Ranking (mayor a menor) ──
";
    ordenarDesc(notas, N);
    mostrarLista(notas, N);

    double buscar;
    cout << "
Nota a buscar: "; cin >> buscar;
    int pos = buscarNota(notas, N, buscar);
    if (pos != -1)
        cout << "Encontrada en posición " << pos+1 << " del ranking
";
    else
        cout << "Nota no encontrada
";
    return 0;
}

void ingresar(double notas[], int n) {
    for (int i = 0; i < n; i++) {
        do {
            cout << "  Nota estudiante [" << i+1 << "]: ";
            cin  >> notas[i];
        } while (notas[i] < 0 || notas[i] > 10);
    }
}

double calcularProm(const double notas[], int n) {
    double s=0; for(int i=0;i<n;i++) s+=notas[i]; return s/n;
}
double calcularMin(const double notas[], int n) {
    double m=notas[0]; for(int i=1;i<n;i++) if(notas[i]<m) m=notas[i]; return m;
}
double calcularMax(const double notas[], int n) {
    double m=notas[0]; for(int i=1;i<n;i++) if(notas[i]>m) m=notas[i]; return m;
}
int contarAprobados(const double notas[], int n) {
    int c=0; for(int i=0;i<n;i++) if(notas[i]>=7) c++; return c;
}
void mostrarLista(const double notas[], int n) {
    for(int i=0;i<n;i++)
        cout<<"  ["<<setw(2)<<i+1<<"] "<<fixed<<setprecision(1)<<notas[i]<<(notas[i]>=7?" ✓":" ✗")<<"
";
}
void ordenarDesc(double notas[], int n) {
    for(int i=0;i<n-1;i++) {
        bool h=false;
        for(int j=0;j<n-1-i;j++)
            if(notas[j]<notas[j+1]){double t=notas[j];notas[j]=notas[j+1];notas[j+1]=t;h=true;}
        if(!h) break;
    }
}
int buscarNota(const double notas[], int n, double obj) {
    for(int i=0;i<n;i++) if(notas[i]==obj) return i;
    return -1;
}
